const int MOD = 1e9 + 7;
const int INF = 1e18;


int32_t main() {
  ios_base::sync_with_stdio(0);
  cin.tie(0);
  

  auto Query = [&](int x) -> bool {
    cout << "? " << x << endl;
    cout.flush();
    string s; cin >> s;
    return s == "YES";
  };

  auto Answer = [&](int x) -> void {
    cout << "! " << x << endl;
    cout.flush();
    return;
  };



  int l = 1, r = 1e9;
  while (l <= r) {
    int mid = (l + r) / 2;
    bool b = Query(mid);
    if (b) l = mid + 1;
    else r = mid - 1;
  }

  Answer(r + 1);

  return 0;
}
