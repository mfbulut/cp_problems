#include <bits/stdc++.h>

using namespace std;

#define int long long
#define endl '\n'

#ifdef LOCAL
#include "debug.h"
#else
#define debug(...) 42
#endif

const int MOD = 1e9 + 7;
const int INF = 1e18;


int32_t main() {
  ios_base::sync_with_stdio(0);
  cin.tie(0);
  
  int TT; cin >> TT;
  while (TT--) {
    int n; cin >> n;

    auto Ask = [&](string s) -> bool {
      cout << "? " << s << endl;
      cout.flush();
      int x; cin >> x;
      return x;
    };

    auto Answer = [&](string s) -> void {
      cout << "! " << s << endl;
      cout.flush();
    };


    
    string s = "0";

    if (!Ask(s)) {
      string ans(n, '1'); // 010110
      Answer(ans);
      continue;
    }

    for (int i = 1; i < 2 * n && s.size() < n; i++) {
      bool a = Ask(s + "0");
      bool b = Ask(s + "1");
      if (!(a | b)) break;
      else if (a) s += "0";
      else s += "1";
    }

    while (s.size() < n) {
      bool a = Ask("0" + s);
      if (a) s = "0" + s;
      else s = "1" + s;
    }

    Answer(s);

  }
  
  return 0;
}
